\begin{theorem}[\href{https://drive.google.com/file/d/1FDjgatdkUNorsutOTGkse4YHox8nUPu1/view}{Slide 1}]
    \label{thm:slide:sd2e}
    Para todo grafo $G = (V, E)$ temos que
    \[
        \sum_{v \in V} \grau_G(v) = 2 \abs{E}
    \]
\end{theorem}

\begin{theorem}[\href{https://drive.google.com/file/d/1OIK_OJCp6hyW0BqqSC6uLjVkKr5Uza1U/view}{Slide 3}]
    \label{thm:slide:mn1}
    Se $G$ é um grafo conexo e acíclico (i.e., uma árvore), então $m(G) = n(G) - 1$.
\end{theorem}
